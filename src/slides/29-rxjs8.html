Exercise time!

Input: 'this', 'is', 'my', 'first', 'exercise'
Exercise: Log the length of every string
Solution:
import { of } from 'rxjs';
import { map } from 'rxjs/operators';
            
of('this', 'is', 'my', 'first', 'exercise')
    .pipe(map((x: string) => x.length))
    .subscribe((x: number) => console.log(x))
