Pipeable operators

eg: map, filter, debounce, buffer, switchMap, ...

const observable = new Observable(subscriber => {
  subscriber.next(0);            
  subscriber.next(1);            
  subscriber.next(2);            
});

observable.pipe(
  map(num => num * 2),
  filter(num => num > 2)
).subscribe(x => console.log(x)); // Output?
4
