Observable Execution

Start transmitting data through the observable.

import { Observable } from 'rxjs';

const observable = new Observable(function subscribe(subscriber) {
   subscriber.next('Hello world!');
   subscriber.complete(); // complete the execution
   subscriber.next('Whut?'); // Will not be send
   subscriber.error(new Error()); // throws error
});

const subscription = observable.subscribe(s => console.log(s));
observable.subscribe({
  next(x) { console.log('got value ' + x); },
  error(err) { console.error('something wrong occurred: ' + err); },
  complete() { console.log('done'); }
});
