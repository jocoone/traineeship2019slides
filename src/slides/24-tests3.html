Unit Testing 

describe('Consultant', () => {
  test('create a consultant', () => {
    const consultant = new Consultant('John Doe', new Date(1990, 1, 1), 'Frontend Consultant');
    expect(consultant.name).toBe('John Doe');
    expect(consultant.birthDate).toEqual(new Date(1990, 1, 1));
    expect(consultant.functionLevel).toBe('Frontend Consultant');
  });
});
