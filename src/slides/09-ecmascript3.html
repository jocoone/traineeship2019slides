Arrow functions

function noArrow() {
    var list = [1, 2, 3, 4, 5];
                                  
    list.filter(function(i) {
        return i % 2 == 0;
    }).map(function(i) {
        return i * 2;
    }).forEach(function(i) {
        console.log(i);
    });
}
function arrow() {
    let list = [1, 2, 3, 4, 5];
                                                                      
    list
        .filter(i => i % 2 == 0)
        .map(i => i * 2)
        .forEach(i => console.log(i));
}
